	public static boolean evaluacionCalidad (int mantenibilidad, int adfuncional){
		
		int[] v = new int[] {mantenibilidad, adfuncional};
		
		int res = min(v);
		
		if (res >= 3) {
			return true;
		} else {
			return false;
		}
		
	}

	public static int min(int [] array) {
	      int min = array[0];
	     
	      for(int i=0; i<array.length; i++ ) {
	         if(array[i]<min) {
	            min = array[i];
	         }
	      }
	      return min;
	}
	
	public static int leerCosas() throws Exception {
	
		int tecto = -1;

		} catch (Exception e) {
			System.out.print("Error al insertar un valor. Fin de programa");
			System.exit(-1);
		}

		
		return tecto;
